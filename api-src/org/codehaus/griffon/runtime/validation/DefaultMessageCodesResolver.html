016 package org.codehaus.griffon.runtime.validation;
017 
018 
019 import griffon.plugins.validation.MessageCodesResolver;
020 
021 import javax.annotation.Nonnull;
022 import java.io.Serializable;
023 import java.util.ArrayList;
024 import java.util.List;
025 
026 /**
027  * Default implementation of the {@link griffon.plugins.validation.MessageCodesResolver} interface.
028  * <p/>
029  * <p>Will create two message codes for an object error, in the following order:
030  * <ul>
031  * <li>1.: code + "." + object name
032  * <li>2.: code
033  * </ul>
034  * <p/>
035  * <p>Will create four message codes for a field specification, in the following order:
036  * <ul>
037  * <li>1.: code + "." + object name + "." + field
038  * <li>2.: code + "." + field
039  * <li>3.: code + "." + field type
040  * <li>4.: code
041  * </ul>
042  * <p/>
043  * <p>For example, in case of code "typeMismatch", object name "user", field "age":
044  * <ul>
045  * <li>1. try "typeMismatch.user.age"
046  * <li>2. try "typeMismatch.age"
047  * <li>3. try "typeMismatch.int"
048  * <li>4. try "typeMismatch"
049  * </ul>
050  * <p/>
051  * <p>This resolution algorithm thus can be leveraged for example to show
052  * specific messages for binding errors like "required" and "typeMismatch":
053  * <ul>
054  * <li>at the object + field level ("age" field, but only on "user");
055  * <li>at the field level (all "age" fields, no matter which object name);
056  * <li>or at the general level (all fields, on any object).
057  * </ul>
058  * <p/>
059  * <p>In case of array, {@link List} or {@link java.util.Map} properties,
060  * both codes for specific elements and for the whole collection are
061  * generated. Assuming a field "name" of an array "groups" in object "user":
062  * <ul>
063  * <li>1. try "typeMismatch.user.groups[0].name"
064  * <li>2. try "typeMismatch.user.groups.name"
065  * <li>3. try "typeMismatch.groups[0].name"
066  * <li>4. try "typeMismatch.groups.name"
067  * <li>5. try "typeMismatch.name"
068  * <li>6. try "typeMismatch.java.lang.String"
069  * <li>7. try "typeMismatch"
070  * </ul>
071  * <p/>
072  * <p>In order to group all codes into a specific category within your resource bundles,
073  * e.g. "validation.typeMismatch.name" instead of the default "typeMismatch.name",
074  * consider specifying a {@link #setPrefix prefix} to be applied.
075  *
076  * @author Juergen Hoeller (Spring 1.0.1)
077  */
078 @SuppressWarnings("serial")
079 public class DefaultMessageCodesResolver implements MessageCodesResolver, Serializable {
080 
081     /**
082      * The separator that this implementation uses when resolving message codes.
083      */
084     public static final String CODE_SEPARATOR = ".";
085 
086 
087     private String prefix = "";
088 
089 
090     /**
091      * Specify a prefix to be applied to any code built by this resolver.
092      * <p>Default is none. Specify, for example, "validation." to get
093      * error codes like "validation.typeMismatch.name".
094      */
095     public void setPrefix(String prefix) {
096         this.prefix = (prefix != null ? prefix : "");
097     }
098 
099     /**
100      * Return the prefix to be applied to any code built by this resolver.
101      * <p>Returns an empty String in case of no prefix.
102      */
103     protected String getPrefix() {
104         return this.prefix;
105     }
106 
107 
108     @Nonnull
109     public String[] resolveMessageCodes(@Nonnull String errorCode, @Nonnull String objectName) {
110         return new String[]{
111             postProcessMessageCode(errorCode + CODE_SEPARATOR + objectName),
112             postProcessMessageCode(errorCode)};
113     }
114 
115     /**
116      * Build the code list for the given code and field: an
117      * object/field-specific code, a field-specific code, a plain error code.
118      * <p>Arrays, Lists and Maps are resolved both for specific elements and
119      * the whole collection.
120      * <p>See the {@link DefaultMessageCodesResolver class level Javadoc} for
121      * details on the generated codes.
122      *
123      * @return the list of codes
124      */
125     @Nonnull
126     public String[] resolveMessageCodes(@Nonnull String errorCode, @Nonnull String objectName, @Nonnull String field, @Nonnull Class<?> fieldType) {
127         List<String> codeList = new ArrayList<String>();
128         List<String> fieldList = new ArrayList<String>();
129         buildFieldList(field, fieldList);
130         for (String fieldInList : fieldList) {
131             codeList.add(postProcessMessageCode(errorCode + CODE_SEPARATOR + objectName + CODE_SEPARATOR + fieldInList));
132         }
133         int dotIndex = field.lastIndexOf('.');
134         if (dotIndex != -1) {
135             buildFieldList(field.substring(dotIndex + 1), fieldList);
136         }
137         for (String fieldInList : fieldList) {
138             codeList.add(postProcessMessageCode(errorCode + CODE_SEPARATOR + fieldInList));
139         }
140         if (fieldType != null) {
141             codeList.add(postProcessMessageCode(errorCode + CODE_SEPARATOR + fieldType.getName()));
142         }
143         codeList.add(postProcessMessageCode(errorCode));
144         return codeList.toArray(new String[codeList.size()]);
145     }
146 
147     /**
148      * Add both keyed and non-keyed entries for the supplied <code>field</code>
149      * to the supplied field list.
150      */
151     protected void buildFieldList(String field, List<String> fieldList) {
152         fieldList.add(field);
153         String plainField = field;
154         int keyIndex = plainField.lastIndexOf('[');
155         while (keyIndex != -1) {
156             int endKeyIndex = plainField.indexOf(']', keyIndex);
157             if (endKeyIndex != -1) {
158                 plainField = plainField.substring(0, keyIndex) + plainField.substring(endKeyIndex + 1);
159                 fieldList.add(plainField);
160                 keyIndex = plainField.lastIndexOf('[');
161             } else {
162                 keyIndex = -1;
163             }
164         }
165     }
166 
167     /**
168      * Post-process the given message code, built by this resolver.
169      * <p>The default implementation applies the specified prefix, if any.
170      *
171      * @param code the message code as built by this resolver
172      * @return the final message code to be returned
173      * @see #setPrefix
174      */
175     protected String postProcessMessageCode(String code) {
176         return getPrefix() + code;
177     }
178 }
